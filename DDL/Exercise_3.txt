
CREATE TABLE Match(
MId INTEGER,
TId INTEGER CONSTRAINT tab_tid_fk REFERENCES TOURNAMENT(TId),
Player1 INTEGER CONSTRAINT tab_pl1_fk REFERENCES PLAYER(PId), 
Player2 INTEGER CONSTRAINT tab_pl2_fk REFERENCES PLAYER(PId),
MatchDt DATE CONSTRAINT tab_mtt_nn NOT NULL,
Winner INTEGER CONSTRAINT tab_wnr_fk REFERENCES PLAYER(PId),
Score VARCHAR2(30)CONSTRAINT tab_sce_nn NOT NULL,
CONSTRAINT tab_id_pk PRIMARY KEY(MId,TId),
CONSTRAINT tab_ply_ck CHECK(Player1 <> Player2)
);
